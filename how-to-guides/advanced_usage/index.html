    Advanced Usage

    This guide covers advanced topics in Sockcan, including loopback modes, extended CAN IDs, filters, and interoperability with python-can.

    1. Configuring Loopback Modes

    Sockcan allows you to control how transmitted messages are looped back to the sending socket or other sockets on the same CAN device.

    Standalone Mode

    from sockcan import SocketcanConfig, connect_to_socketcan
from sockcan._protocol import LoopbackMode, build_send_func, build_recv_func
import time

# Example 1: LoopbackMode.FOR_OTHER_SOCKS (Default)
# Messages are received by other sockets, but not the sender.
print("\n--- LoopbackMode.FOR_OTHER_SOCKS (Default) ---")
config_for_others = SocketcanConfig(channel="vcan0", loopback=LoopbackMode.FOR_OTHER_SOCKS)
sock1 = connect_to_socketcan(config_for_others)
sock2 = connect_to_socketcan(config_for_others)

try:
    send_func1 = build_send_func(sock1, expects_msg_cls=False)
    recv_func1 = build_recv_func(sock1)
    recv_func2 = build_recv_func(sock2)

    message_data = b'\x11\x22'
    send_func1(0x100, message_data, False)
    print("Sent message with ID 0x100 from sock1.")

    # sock2 SHOULD receive the message
    received_by_sock2 = recv_func2()
    print(f"Sock2 received: ID={received_by_sock2.arbitration_id:#x}, Data={received_by_sock2.data.hex()}")

finally:
    sock1.close()
    sock2.close()

# Example 2: LoopbackMode.ON
# Messages are received by the sending socket itself.
print("\n--- LoopbackMode.ON ---")
config_on = SocketcanConfig(channel="vcan0", loopback=LoopbackMode.ON)
sock_loopback_on = connect_to_socketcan(config_on)

try:
    send_func_on = build_send_func(sock_loopback_on, expects_msg_cls=False)
    recv_func_on = build_recv_func(sock_loopback_on)

    message_data = b'\xAA\xBB'
    send_func_on(0x200, message_data, False)
    print("Sent message with ID 0x200 from sock_loopback_on.")

    received_self = recv_func_on()
    print(f"Sock_loopback_on received its own message: ID={received_self.arbitration_id:#x}, Data={received_self.data.hex()}")

finally:
    sock_loopback_on.close()

# Example 3: LoopbackMode.OFF
# Messages are not received by any socket (sender or others).
print("\n--- LoopbackMode.OFF ---")
config_off = SocketcanConfig(channel="vcan0", loopback=LoopbackMode.OFF)
sock_loopback_off = connect_to_socketcan(config_off)
sock_other = connect_to_socketcan(config_off)

try:
    send_func_off = build_send_func(sock_loopback_off, expects_msg_cls=False)
    recv_func_other = build_recv_func(sock_other)

    message_data = b'\xCC\xDD'
    send_func_off(0x300, message_data, False)
    print("Sent message with ID 0x300 from sock_loopback_off.")
    print("Neither socket should receive the message in LoopbackMode.OFF")

finally:
    sock_loopback_off.close()
    sock_other.close()

    python-can Interoperability Mode

    Loopback can be configured through python-can's receive_own_messages parameter:

    import can
from sockcan.interop import hijack_python_can

hijack_python_can()

# receive_own_messages=True corresponds to LoopbackMode.ON
bus = can.Bus(interface="socketcan", channel="vcan0", receive_own_messages=True)

    2. Sending Extended CAN ID Messages

    Extended CAN IDs (29-bit) are supported by setting the is_extended flag to True.

    Standalone Mode

    from sockcan import SocketcanConfig, connect_to_socketcan
from sockcan._protocol import build_send_func, build_recv_func

config = SocketcanConfig(channel="vcan0")
sock_tx = connect_to_socketcan(config)
sock_rx = connect_to_socketcan(config)

try:
    send_func = build_send_func(sock_tx, expects_msg_cls=False)
    recv_func = build_recv_func(sock_rx)

    extended_arbitration_id = 0x18DA00F1  # Common extended ID example (OBD-II)
    extended_data = b'\x0A\x0B\x0C\x0D\x0E\x0F\x10\x11'
    is_extended = True

    send_func(extended_arbitration_id, extended_data, is_extended)
    print(f"Sent extended message: ID={extended_arbitration_id:#x}, Data={extended_data.hex()}")

    received_message = recv_func()
    print("Received Extended Message:")
    print(f"  Arbitration ID: {received_message.arbitration_id:#x}")
    print(f"  Data: {received_message.data.hex()}")
    print(f"  Is Extended ID: {received_message.is_extended_id}")

finally:
    sock_tx.close()
    sock_rx.close()

    python-can Interoperability Mode

    import can
from sockcan.interop import hijack_python_can

hijack_python_can()

bus = can.Bus(interface="socketcan", channel="vcan0")

# Extended ID message
msg = can.Message(
    arbitration_id=0x18DA00F1,
    data=[0x02, 0x01, 0x0D, 0x00, 0x00, 0x00, 0x00, 0x00],
    is_extended_id=True,
)
bus.send(msg)
print(f"Sent extended ID: {msg.arbitration_id:#x}")

    3. Filtering Messages

    Standalone Mode

    Use the SocketcanServer.subscribe() method with filter sets:

    from sockcan import SocketcanConfig, connect_to_socketcan
from sockcan._protocol import build_send_func, build_recv_func
from sockcan.daemon import SocketcanServer

# Create a server with message filtering
server = SocketcanServer(use_stream=True)
server.start()

# Subscribe with filters - only receive messages with IDs 0x100-0x1FF
sock = server.subscribe(filters={0x100, 0x101, 0x102, 0x200})

recv_func = build_recv_func(sock)
send_func = build_send_func(sock, expects_msg_cls=False)

# These will be received
send_func(0x100, b'\x01\x02', False)
msg = recv_func()
print(f"Received filtered: ID={msg.arbitration_id:#x}")

# This will be filtered out
send_func(0x999, b'\x03\x04', False)
# msg = recv_func()  # Would block - message was filtered

    python-can Interoperability Mode

    Use python-can's can_filters parameter:

    import can
from sockcan.interop import hijack_python_can

hijack_python_can()

# Filter to only receive messages with IDs 0x100-0x1FF
bus = can.Bus(
    interface="socketcan",
    channel="vcan0",
    can_filters=[{"can_id": 0x100, "can_mask": 0x700}],  # ID range 0x100-0x1FF
)

msg = bus.recv()
print(f"Received: ID={msg.arbitration_id:#x}")

    4. Using the Daemon for Concurrent Access

    For interfaces that don't support concurrent access (like PCAN on Windows), use the SocketcanServer to multiplex connections:

    Standalone Mode

    from sockcan.daemon import SocketcanServer, BusParameters
from sockcan import build_send_func, build_recv_func

# Create a server wrapping real hardware
with SocketcanServer.factory(
    BusParameters(channel="PCAN_USBBUS1", interface="pcan", bitrate=500000)
) as server:
    # Multiple sockets can subscribe
    sock1 = server.subscribe()
    sock2 = server.subscribe()

    send1 = build_send_func(sock1, expects_msg_cls=False)
    send2 = build_send_func(sock2, expects_msg_cls=False)
    recv1 = build_recv_func(sock1)
    recv2 = build_recv_func(sock2)

    # Both sockets can send and receive concurrently
    send1(0x100, b'\x01\x02', False)
    send2(0x200, b'\x03\x04', False)

    # Both receive all messages
    msg1 = recv1()
    msg2 = recv2()
    print(f"Sock1 received: {msg1.arbitration_id:#x}")
    print(f"Sock2 received: {msg2.arbitration_id:#x}")

    python-can Interoperability Mode

    Use activate_userspace_socketcan():

    import can
from sockcan.interop import activate_userspace_socketcan, SocketcanDaemonConfig
from sockcan.daemon import BusParameters

config = SocketcanDaemonConfig(mode="local")
activate_userspace_socketcan(
    BusParameters(channel="PCAN_USBBUS1", interface="pcan", bitrate=500000),
    config=config,
)

# Multiple python-can buses can now share the same hardware
bus1 = can.Bus(interface="socketcan", channel="PCAN_USBBUS1")
bus2 = can.Bus(interface="socketcan", channel="PCAN_USBBUS1")

# Both buses share the same hardware concurrently
bus1.send(can.Message(arbitration_id=0x100, data=[1, 2]))
bus2.send(can.Message(arbitration_id=0x200, data=[3, 4]))

    5. Virtual vs Real CAN Interface

    All examples in this guide use vcan0 (virtual CAN). To use real hardware:

    Standalone Mode

    # Virtual CAN
config = SocketcanConfig(channel="vcan0")

# Real CAN hardware
config = SocketcanConfig(channel="can0")

    python-can Mode

    # Virtual CAN
bus = can.Bus(interface="socketcan", channel="vcan0")

# Real CAN hardware
bus = can.Bus(interface="socketcan", channel="can0")

    Daemon Mode with Real Hardware

    from sockcan.daemon import SocketcanServer, BusParameters

# Linux native socketcan
with SocketcanServer.factory(
    BusParameters(channel="can0", interface="socketcan")
) as server:
    sock = server.subscribe()

# PCAN on Windows
with SocketcanServer.factory(
    BusParameters(channel="PCAN_USBBUS1", interface="pcan", bitrate=500000)
) as server:
    sock = server.subscribe()

    6. Context Manager Usage

    Both standalone and interoperability modes support context managers for automatic cleanup:

    Standalone Mode

    from sockcan import SocketcanConfig, connect_to_socketcan, build_send_func, build_recv_func

with connect_to_socketcan(SocketcanConfig(channel="vcan0")) as sock:
    send = build_send_func(sock, expects_msg_cls=False)
    recv = build_recv_func(sock)

    send(0x100, b'\x01\x02', False)
    msg = recv()
    print(f"Received: {msg.arbitration_id:#x}")

# Socket automatically closed on exit

    python-can Mode

    import can
from sockcan.interop import hijack_python_can

hijack_python_can()

with can.Bus(interface="socketcan", channel="vcan0") as bus:
    msg = can.Message(arbitration_id=0x100, data=[1, 2])
    bus.send(msg)
    response = bus.recv()

# Bus automatically shutdown on exit
